=== STATE LOGIC (verbatim from mockup's inline JS — port to React useState) ===
const setState = (s, wallet) => {
  if (s === 'connecting') {
    modal.classList.add('state-connecting');
    title.textContent = 'Connecting';
    if (wallet) {
      connTitle.textContent = `Opening ${wallet}…`;
      const map = {
        'MetaMask':         ['icon-metamask','M'],
        'Coinbase Wallet':  ['icon-coinbase','C'],
        'WalletConnect':    ['icon-wc','W'],
        'Rainbow':          ['icon-rainbow','R'],
        'Browser wallet':   ['icon-injected','⌘'],
      };
      const [cls, ch] = map[wallet] || ['icon-metamask','M'];
      connIcon.className = 'conn-icon ' + cls;
      connIcon.textContent = ch;
    }
  } else {
    modal.classList.remove('state-connecting');
    title.textContent = 'Connect a wallet';
  }
};

// Wallet buttons trigger state change
document.querySelectorAll('.wallet-btn').forEach(btn => {
  btn.addEventListener('click', () => setState('connecting', btn.dataset.wallet));
});

// Back button returns to default
back.addEventListener('click', () => { setState('default'); syncToggle('default'); });
document.getElementById('pick-another').addEventListener('click', () => {
  setState('default'); syncToggle('default');
});

// Scrim toggle (only for preview in the mockup — real React impl won't need this)
toggle.addEventListener('click', e => {
  const b = e.target.closest('button');
  if (!b) return;
  setState(b.dataset.state, 'MetaMask');
  syncToggle(b.dataset.state);
});
